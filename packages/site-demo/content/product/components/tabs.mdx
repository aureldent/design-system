```javascript import
import { useState } from 'react';
import classNames from 'classnames';
import { Button, Tab, Tabs, TabsLayout, Theme } from '@lumx/react';
import { mdiBowl, mdiBreadSliceOutline, mdiSilverwareForkKnife } from '@lumx/icons';
```

# Tabs

**Tabs are navigation elements that regroup related sections of content for users to navigate.**

## Full width

```javascript jsx withThemeSwitcher disableGrid
(theme) => {
    const [activeTab, setActiveTab] = useState(0);

    const handleTabClick = ({ index }) => {
        setActiveTab(index);
    };

    return (
        <>
            <Tabs theme={theme} activeTab={activeTab} onTabClick={handleTabClick}>
                <Tab label="Tab 1">
                    <p
                        className={classNames(
                            'lumx-spacing-padding-vertical-huge',
                            theme === Theme.dark && 'lumx-theme-color-light-N',
                        )}
                    >
                        Tab 1 content
                    </p>
                </Tab>

                <Tab label="Tab 2" isDisabled={true}>
                    <p
                        className={classNames(
                            'lumx-spacing-padding-vertical-huge',
                            theme === Theme.dark && 'lumx-theme-color-light-N',
                        )}
                    >
                        Tab 2 content
                    </p>
                </Tab>

                <Tab label="Tab 3">
                    <p
                        className={classNames(
                            'lumx-spacing-padding-vertical-huge',
                            theme === Theme.dark && 'lumx-theme-color-light-N',
                        )}
                    >
                        Tab 3 content
                    </p>
                </Tab>
            </Tabs>
        </>
    );
};
```
